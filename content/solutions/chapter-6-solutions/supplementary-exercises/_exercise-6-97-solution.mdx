$$
\begin{align*}
    p_W(w) & = \sum_{y_1 = 0}^w p_{Y_1, Y_2}(y_1, w - y_1)\\
    & = \sum_{y_1 = 0}^w p_{Y_1}(y_1) p_{Y_2}(w - y_1)\\
    & = \sum_{y_1 = 0}^w \binom{4}{y_1} \left(\frac{1}{5} \right)^{y_1} \left(\frac{4}{5} \right)^{4 - y_1} \binom{3}{w - y_1} \left(\frac{1}{2} \right)^{w - y_1} \left(\frac{1}{2} \right)^{3 - (w - y_1)}\\

\end{align*}
$$
