a) Letting $F_{Y_1}(y_1) = F_{Y_2}(y_2) = F(y)$, then

$$
\begin{align*}
    f_{U_1}(u_1) & = n[1 - F(u_1)]^{n - 1} f(u_1)\\
    & = 2 \left[1 -
    \begin{cases}
        0, & u_1 < 0,\\
        u_1, & 0 \leq u_1 \leq 1,\\
        1, & 1 < u_1
    \end{cases} \right]^{2 - 1} \cdot
    \begin{cases}
        1, & 0 \leq u_1 \leq 1,\\
        0, & \text{elsewhere}
    \end{cases}\\
    & =
    \begin{cases}
        2(1-  u_1), & 0 \leq u_1 \leq 1,\\
        0, & \text{elsewhere.}
    \end{cases}
\end{align*}
$$

b)

$$
\begin{align*}
    E(U_1^m) & = \int_{u_1 = 0}^{u_1 = 1} u_1^m \cdot 2(1 - u_1) du_1\\
    & = \left[\frac{2}{m + 1} u_1^{m + 1} - \frac{2}{m + 2} u_1^{m + 2} \right]_{u_1 = 0}^{u_1 = 1}\\
    & = \left(\frac{2}{m + 1} (1)^{m + 1} - \frac{2}{m + 2} (1)^{m + 2} \right) - \left(\frac{2}{m + 1} (0)^{m + 1} - \frac{2}{m + 2} (0)^{m + 2} \right)\\
    & = \frac{2}{m + 1} - \frac{2}{m + 2} = \frac{2(m + 2) - 2(m + 1)}{(m + 1)(m + 2)} = \frac{2}{(m + 1)(m + 2)}\\
    \implies E(U_1) & = E(U_1^1) = \frac{2}{((1) + 1)((1) + 2)} = \frac{2}{6} = \frac{1}{3}\\
    E(U_1^2) & = \frac{2}{((2) + 1)((2) + 2)} = \frac{2}{12} = \frac{1}{6}\\
    V(U_1) & = E(U_1^2) - E(U_1)^2 = \frac{1}{6} - \left(\frac{1}{3} \right)^2 = \frac{3}{18} - \frac{2}{18} = \frac{1}{18}
\end{align*}
