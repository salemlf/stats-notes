a)

$$
\begin{align*}
    f_{Y_{(k)}}(y_k) & = \frac{n!}{(k - 1)! \, (n - k)!} [F(y_k)]^{k - 1} [1 - F(y_k)]^{n - k} f(y_k)\\
    & = \frac{n!}{(k - 1)! \, (n - k)!} \left[
    \begin{cases}
        0, & y_k < 0,\\
        \frac{y_k}{\theta}, & 0 \leq y_k \leq \theta,\\
        1, & \theta < y_k
    \end{cases} \right]^{k - 1}\\
    & \times \left[1 -
    \begin{cases}
        0, & y_k < 0,\\
        \frac{y_k}{\theta}, & 0 \leq y_k \leq \theta,\\
        1, & \theta < y_k
    \end{cases} \right]^{n - k} \cdot
    \begin{cases}
        \frac{1}{\theta}, & 0 \leq y_k \leq \theta,\\
        0, & \text{elsewhere}
    \end{cases}\\
    & =
    \begin{cases}
        \frac{n!}{(k - 1)! \, (n - k)!} \cdot \left(\frac{y_k}{\theta} \right)^{k - 1} \left(1 - \frac{y_k}{\theta} \right)^{n - k} \cdot \frac{1}{\theta}\\
        = \frac{1}{\theta} \cdot \frac{\Gamma(n + 1)}{\Gamma(k) \Gamma(n - (k - 1))} \left(\frac{y_k}{\theta} \right)^{k - 1} \left(1 - \frac{y_k}{\theta} \right)^{(n - (k - 1)) - 1}\\
        = \frac{1}{\theta} \cdot \frac{\left(\frac{y_k}{\theta} \right)^{k - 1} \left(1 - \frac{y_k}{\theta} \right)^{(n - (k - 1)) - 1}}{B(k, n - (k - 1))}, & 0 \leq y_k \leq \theta,\\
        0, & \text{elsewhere}
    \end{cases}
\end{align*}
$$

---

b)

$$
\begin{align*}
    E(Y_{(k)}) & = \int_{y_k = 0}^{y_k = \theta} \frac{y_k}{\theta} \cdot \frac{\left(\frac{y_k}{\theta} \right)^{k - 1} \left(1- \frac{y_k}{\theta} \right)^{(n - (k - 1)) - 1}}{B(k, n - (k - 1))} dy_k\\
    & = \theta \int_{u = 0}^{u = 1} u \cdot \frac{u^{k - 1} \left(1 - u \right)^{(n - (k - 1)) - 1}}{B(k, n - (k - 1))} du \quad (\text{letting $u = y_k / \theta$ so $du = dy_k / \theta$})\\
    & = \theta \cdot \frac{k}{k + (n - (k - 1))} = \frac{k \theta}{n + 1}
\end{align*}
$$

---

c)

$$
\begin{align*}
    E(Y_{(k)}^2) & = \int_{y_k = 0}^{y_k = \theta} \frac{y_k^2}{\theta} \cdot \frac{\left(\frac{y_k}{\theta} \right)^{k - 1} \left(1- \frac{y_k}{\theta} \right)^{(n - (k - 1)) - 1}}{B(k, n - (k - 1))} dy_k\\
    & = \int_{y_k = 0}^{y_k = \theta} \theta \cdot \left(\frac{y_k}{\theta} \right)^2 \cdot \frac{\left(\frac{y_k}{\theta} \right)^{k - 1} \left(1- \frac{y_k}{\theta} \right)^{(n - (k - 1)) - 1}}{B(k, n - (k - 1))} dy_k\\
    & = \theta^2 \int_{u = 0}^{u = 1} u^2 \cdot \frac{u^{k - 1} (1- u)^{(n - (k - 1)) - 1}}{B(k, n - (k - 1))} du \quad (\text{letting $u = y_k / \theta$ so $du = dy_k / \theta$})\\
    & = \theta^2 \left[\frac{k(n - (k - 1))}{(n + 1)^2 (n + 2)} + \left(\frac{k}{n + 1} \right)^2 \right]\\
    & (\text{the integral equals $E(U^2) = V(U) + E(U)^2$ for a beta random}\\
    & \text{variable $U$ with $\alpha = k$ and $\beta = n - (k - 1)$})\\
    \implies V(Y_{(k)}) & = E(Y_{(k)}^2) - E(Y_{(k)})^2\\
    & = \theta^2 \left[\frac{k(n - (k - 1))}{(n + 1)^2 (n + 2)} + \left(\frac{k}{n + 1} \right)^2 \right] - \left(\frac{k \theta}{n + 1} \right)^2\\
    & = \frac{k(n - k + 1) \theta^2}{(n + 1)^2 (n + 2)} + \left(\frac{k}{n + 1} \right)^2 \theta^2 - \left(\frac{k}{n + 1} \right)^2 \theta^2\\
    & = \frac{k(n - k + 1) \theta^2}{(n + 1)^2 (n + 2)}
\end{align*}
$$

d)

$$
E(Y_{(k)} - Y_{(k - 1)}) = E(Y_{(k)}) - E(Y_{(k - 1)}) = \frac{k \theta}{n + 1} - \frac{(k - 1) \theta}{n + 1} = \frac{\theta}{n + 1}
$$

Since this does not depend on $k$, then the expected difference between each successive pair of order statistics will be the same.
